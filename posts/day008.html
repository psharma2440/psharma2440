goLanguage Program - Using Nested if-else statement


Nested if else statement: When one if-else structure is embedded into another if-else it is called nested if-else.



"Write a Program to take an input from user and check the following conditions Using Nested if-else:"
a) if temp is equal to 30 then print string - It`s lovely day!
b) else - if temp is greater than 30 then print string - It`s cold day!
b) else - if temp is less than 30 then print string - It`s sunny day!
d) else print the string - Enjoy your day! 

            
                package main

import (
	"bufio"
	"fmt"
	"os"
	"strconv"
	"strings"
	"time"
)

func main() {
	reader := bufio.NewReader(os.Stdin)
	fmt.Println("Enter temp")
	temp, _ := reader.ReadString('\n')
	temp = strings.TrimSpace(temp)
	temp1, _ := strconv.Atoi(temp)
	fmt.Println("Temp entered by user:", temp1)
	if temp1 == 30 {
		fmt.Println("It`s lovely day!") // first statement
	} else if temp1 > 30 {
		fmt.Println("It`s sunny day!") // second statement
	} else if temp1 < 30 {
		fmt.Println("It`s cold day!") // third statement
	} else {
		fmt.Println("Enjoy your day!") // fourth statement
	}
}
