Program 06 - Using for loop

Write a program to enter the string from user and then print the string 10 times using for loop. 

            
                package main

//print user name 10 times
import (
	"bufio"
	"fmt"
	"os"
	"strings"
)

func main() {
	reader := bufio.NewReader(os.Stdin)
	fmt.Println("Enter the string:")
	string, _ := reader.ReadString('\n')
	string = strings.TrimSpace(string)
	fmt.Println("string enter by user:", string)
	var i int = 0
	for ; i <10; i++ {
		fmt.Println("i =", i,"String:" ,string)
	}
	fmt.Println("String print",i,"times.")
}
