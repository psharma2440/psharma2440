Program 3 - Array

Write a program to print 10 float numbers with their Index value and numbers are present in an aaray.

        
package main

import "fmt"

func main() {
	var numbers [10]float32 = [10]float32{10.1, 20.1, 2.3, 122.01, 1.2, 23.1, 11.2, 23.4, 26.8, 13.7}
	fmt.Println("10 float numbers are:", numbers)
	for index, value := range numbers {
		fmt.Println("Index:", index, "Value:", value)
	}
}
