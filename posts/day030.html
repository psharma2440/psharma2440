Program 9 - Array

Write a program to find out:-

1)Even Number
2)Odd Number
3)Zero`s Number
4)Negative Number
5)Positive Number

        
package main

import (
	"bufio"
	"fmt"
	"os"
	"strconv"
	"strings"
)

func main() {
	reader := bufio.NewReader(os.Stdin)
	var numbers [5]int
	counter := 0
	for ; counter < 5; counter++ {
		fmt.Println("Enter the value of a", counter+1, "number:")
		number, _ := reader.ReadString('\n')
		number = strings.TrimSpace(number)
		intNumbers, _ := strconv.Atoi(number)
		numbers[counter] = intNumbers
	}
	for index, value := range numbers {
		if value%2 == 0 {
			fmt.Println("The even number of index", index, "and the value is:", value)
		}
		if value%2 != 0 {
			fmt.Println("The odd number of index", index, " and the value is:", value)
		}
	}
	for index, value := range numbers {
		if value > 0 {
			fmt.Println("The positive number of index ", index, "and the value is:", value)
		} else if value < 0 {
			fmt.Println("The negative number of index", index, " and the value is:", value)
		}
		if value == 0 {
			fmt.Println("The zero number of index", index, "and the value is:", value)
		}

	}
}

Explanation:

In this program, we find the positive number, negative number, even number, odd number, zero`s number.
