~ ~ ~ ~ GO Program "To take two numbers from user and print the sum of two numbers".~ ~ ~ ~ 

                
                // Take inputs from user and sum them.



                package main

                import (
	            "bufio"
	            "fmt"
	            "os"
                "strconv"
                "strings"
                )
                func main() {
                    reader := bufio.NewReader(os.Stdin)
                    fmt.Println("Enter a number:")
                    number1, _ := reader.ReadString('\n')
                    number1 = strings.TrimSpace(number1)
                    intNumber1, _ := strconv.Atoi(number1)
                    fmt.Println("Enter another number:")
                    number2, _ := reader.ReadString('\n')
                    number2 = strings.TrimSpace(number2)
                    intNumber2, _ := strconv.Atoi(number2)
                    sum := intNumber1 + intNumber2
                    fmt.Println("Sum of number1 and number2:", sum)
                }
            
        
            
                 // Output:
                 Enter a number:
                 1
                 Enter another number:
                 2
                 Sum of number1 and number2: 3

//Area of Rectangle

package main

// Area of rectangle
import (
	"bufio"
	"fmt"
	"os"
	"strconv"
	"strings"
)

func main() {
	reader := bufio.NewReader(os.Stdin)
	fmt.Println("Enter the length")
	length, _ := reader.ReadString('\n')
	length = strings.TrimSpace(length)
	intLength, _ := strconv.Atoi(length)
	fmt.Println("Enter the breadth")
	breadth, _ := reader.ReadString('\n')
	breadth = strings.TrimSpace(breadth)
	intBreadth, _ := strconv.Atoi(breadth)
	area := intLength * intBreadth
	fmt.Println("Area ", area)
}

                // Check user is eligible for vote or not.
                package main

import (
	"bufio"
	"fmt"
	"os"
	"strconv"
	"strings"
)

func main() {
	reader := bufio.NewReader(os.Stdin)
	fmt.Println("Enter age:")
	age, _ := reader.ReadString('\n')
	age = strings.TrimSpace(age)
	intAge, _ := strconv.Atoi(age)
	fmt.Println("Age enter by user:", intAge)
	if intAge < 18 {
		fmt.Println("You are not eligible for vote.")
	}
	if intAge >= 18 {
		fmt.Println("You are eligible for vote")
	}

}
