Program - 5

Write a program to calculate the frequency of the number.

        
            package main

            import (
                "bufio"
                "fmt"
                "os"
                "strconv"
                "strings"
            )
            
            func main() {
                reader := bufio.NewReader(os.Stdin)
                var arr [5]int
                fmt.Println("Enter the elements in arr:")
                i := 0
                for ; i < 5; i++ {
                    fmt.Println("Element value at", i+1)
                    value, _ := reader.ReadString('\n')
                    value = strings.TrimSpace(value)
                    intValue, _ := strconv.Atoi(value)
                    arr[i] = intValue
                }
                fmt.Println("Elements in arr array are:", arr[0:5])
                freq := 0
                fmt.Println("Enter the no. to find out the frequency")
                num, _ := reader.ReadString('\n')
                num = strings.TrimSpace(num)
                intNum, _ := strconv.Atoi(num)
                for i := 0; i < 5; i++ {
                    if arr[i] == intNum {
                        freq++
                    }
                }
                fmt.Println("Frequence of", intNum, "is", freq)
            }
