Summary of Array.

Part - 1

How to insert elements in Array and display the elements as output?

    
        package main

import (
	"fmt"
)

func main() {
	var arr [6]int = [6]int{1, 2, 3, 4, 10, 11}
	fmt.Println(arr)
}

Output:

    
        [1 2 3 4 10 11]

Explanation:

In this program we insert the elements in an Array(arr).
And as an output we print the elements that are present in arr.



How to insert an element in particular position?

    
        package main

import (
	"fmt"
)

func main() {
	var arr [6]int
	arr[1] = 34
	arr [2] = 45
	fmt.Println(arr)
}

Output:-

    
        [0 34 45 0 0 0]

Explanation:

In this program, we insert the elements at two position i.e.,
At index[1] we insert the element 34.
At index[2] we insert the element 45.
and then print all the elements of array (arr) as an output.



How to user insert the values in array?
or
How to take inputs from user?

         
            package main

            import (
                "bufio"
                "fmt"
                "os"
                "strconv"
                "strings"
            )
            
            func main() {
                reader := bufio.NewReader(os.Stdin)
                var arr [4]int
                fmt.Println("Please insert the values in arr")
                for c := 0; c < 4; c++ {
                    fmt.Println("Value at", c+1, "element:")
                    values, _ := reader.ReadString('\n')
                    values = strings.TrimSpace(values)
                    intValues, _ := strconv.Atoi(values)
                    arr[c] = intValues
                }
                fmt.Println("Elements in arr are:", arr)
            }

Output:

        
Please insert the values in arr
Value at 1 element:
23
Value at 2 element:
34
Value at 3 element:
44
Value at 4 element:
55
Elements in arr are: [23 34 44 55]

Explanation:

In this program, user insert the values in arr upto 3th index.

and it is followed by the name of the variable (arr) we want to loop over.



Type conversion in Array.

    
        package main

import (
	"bufio"
	"fmt"
	"os"
	"strconv"
	"strings"
)

func main() {
	reader := bufio.NewReader(os.Stdin)
	var arr [2]int
	fmt.Println("Please insert the values in arr:")
	var sum float64 = 0
	for c := 0; c < 2; c++ {
		fmt.Println("Value at", c+1, "element:")
		values, _ := reader.ReadString('\n')
		values = strings.TrimSpace(values)
		intValues, _ := strconv.Atoi(values)
		arr[c] = intValues
        sum = sum + arr[c] // error
	}
	fmt.Println("Elements in arr are:", arr)
	fmt.Println("Avearge:", sum/len(arr)) // error 
	for i, v := range arr {
		fmt.Println("Index", i, "Value", v)
	}
}

Output:

    
        invalid operation: sum + arr[c] (mismatched types float64 and int)
        invalid operation: sum / len(arr) (mismatched types float64 and int)

Explanation:

In this program, Errors occur i.e.
1) invalid operation: sum + arr[c] (mismatched types float64 and int)
2)invalid operation: sum / len(arr) (mismatched types float64 and int)
To remove these errors we should do type conversion means in this we have to change the type of len(arr) and arr[c] int to float64.
Type conversion:
sum = sum + float64(arr[c])
fmt.Println("Avearge:", sum/float64(len(arr)))
