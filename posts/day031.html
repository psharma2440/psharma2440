Program 10 - Array

Write a program to find out the average marks of the 5 students which learn Operating system subject.

        
            package main

import (
	"bufio"
	"fmt"
	"os"
	"strconv"
	"strings"
)

func main() {
	reader := bufio.NewReader(os.Stdin)
	fmt.Println("Operating System Sessional marks of 5 students")
	var students [5]int
	sum := 0
	for counter := 0; counter < 5; counter++ {
		fmt.Println("Roll NO.", counter+1, "marks:")
		marks, _ := reader.ReadString('\n')
		marks = strings.TrimSpace(marks)
		intMarks, _ := strconv.Atoi(marks)
		students[counter] = intMarks
		sum = sum + intMarks
	}
	var averageMarks int = sum / 5
	fmt.Println("Average marks:",averageMarks)
}

Explanation:-

In this program, we find out the average marks of the 5 students which learn Operating system subject.
We take the marks of 5 students from user as an input.
And then as an output we calculate the average marks of the 5 students.
