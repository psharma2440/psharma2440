Program 8- Array

Write a program to take 10 integer inputs from user and store them in an array.
Again ask user to give a number.
Now, tell user whether that number is present in array or not.

        
            package main

import (
	"bufio"
	"fmt"
	"os"
	"strconv"
	"strings"
)

func main() {
	reader := bufio.NewReader(os.Stdin)
	var integerNumber [5]int
	for counter := 0; counter < 5; counter++ {
		fmt.Println("Integer number", counter+1, ":")
		numbers, _ := reader.ReadString('\n')
		numbers = strings.TrimSpace(numbers)
		intNumbers, _ := strconv.Atoi(numbers)
		integerNumber[counter] = intNumbers
	}
	fmt.Println("5 interger numbers are:", integerNumber)
	
	fmt.Println("Enter a number:")
		usernumber, _ := reader.ReadString('\n')
		usernumber = strings.TrimSpace(usernumber)
		intuserNumber, _ := strconv.Atoi(usernumber)
		fmt.Println("Number entered by user:", intuserNumber)
		for index, value := range integerNumber {
			if intuserNumber == value {
				fmt.Println("index:", index, "value:", value)
			fmt.Println(intuserNumber, "Number is present in array")
		}

	}

}

Explanation:

In this program we take 5 interger from user as an input and store them in an array.
And then again ask the user to enter a number,
Then check wheter a number is present is an array or not.
As an output we tell user whether that number is present in array or not.
