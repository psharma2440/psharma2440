Program 6 - Array

Write a program to find out the smallest number in an array and take the number from user as an input.

        
            package main

import (
	"bufio"
	"fmt"
	"os"
	"strconv"
	"strings"
)

func main() {
	reader := bufio.NewReader(os.Stdin)
	var a [4]int
	fmt.Println("Enter the four number")
	for counter := 0; counter < 4; counter++ {
		fmt.Println("Value of", counter+1, "number:")
		number, _ := reader.ReadString('\n')
		number = strings.TrimSpace(number)
		intNumber, _ := strconv.Atoi(number)
		a[counter] = intNumber
	}
	var small int 
	small = a[0]
	for index, value := range a {
		fmt.Println("Index:", index, "Value:", value)
		if value < small {
			small = value
		}
	}
	fmt.Println("Smallest number:", small)
}

In this program, we take the four numbers from user as an input.
and then we find the smallest number between them.
