Creating a slice with make.

        
            package main
            
            import (
                "bufio"
                "fmt"
                "os"
                "strconv"
                "strings"
            )
            
            func main() {
                reader := bufio.NewReader(os.Stdin)
                mySlice := make([]int, 1, 100)
                fmt.Println("Enter the number:")
                num, _ := reader.ReadString('\n')
                num = strings.TrimSpace(num)
                intNum, _ := strconv.Atoi(num)
                for counter := 0; counter < intNum; counter++ {
                    fmt.Println("Value at",counter+1,":")
                    numbers, _ := reader.ReadString('\n')
                    numbers = strings.TrimSpace(numbers)
                    intNumber, _ := strconv.Atoi(numbers)
                    mySlice = append(mySlice, intNumber)
                    // mySlice[counter] = intNumber
                }
            
                reverseSlice := make([]int, 0, 100)
                for i := len(mySlice) - 1; i > 0; i-- {
                    // reverseSlice[j] = mySlice[i-1]
                    reverseSlice = append(reverseSlice, mySlice[i])
                }
                fmt.Println("Reverse of slice:")
                for i := 0; i < len(reverseSlice); i++ {
                    fmt.Println(reverseSlice[i])
                }
            }
