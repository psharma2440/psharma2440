Write a program to count the people who can vote, age of people are present in an array.

        
            package main
            
            import (
                "fmt"
            )
            
            func main() {
                var a [10]int = [10]int{19, 29, 8, 10, 88, 54, 16, 54, 77, 42}
                count := 0
                for i:=0 ; i < 10; i++ {
                    if a[i] >= 18 {
                        fmt.Println("People", a[i],"age is eligble for vote.")
                        count +=1
                    }
                }
                fmt.Println("Total", count, "people can vote out of 10.")
            }
