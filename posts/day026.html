Program 5 - Array

Write a program to take 5 integer inputs from user and store them in an array.
Now, copy all the elements in an another array but in reverse order.

            
                package main

import (
	"bufio"
	"fmt"
	"os"
	"strconv"
	"strings"
)

func main() {
	reader := bufio.NewReader(os.Stdin)
	fmt.Println("Enter 5 numbers in an array.")
	var arr [5]int
	for counter := 0; counter < 5; counter++ {
		fmt.Println("Number", counter+1, "in an aaray:")
		number, _ := reader.ReadString('\n')
		number = strings.TrimSpace(number)
		intNumber, _ := strconv.Atoi(number)
		arr[counter] = intNumber

	}
	var reversedArr [5]int
	var j int
	for i := len(arr); i > 0; i-- {
		reversedArr[j] = arr[i-1]
		j++
	}
	fmt.Println("Numbers in reverser order :-", reversedArr)
}
