Program 09 - Using for loop

Write a program to find the cube of a number.

            
package main

import (
	"bufio"
	"fmt"
	"os"
	"strconv"
	"strings"
)

func main() {
	reader := bufio.NewReader(os.Stdin)
	var cube int
	fmt.Println("Enter a number:")
	number, _ := reader.ReadString('\n')
	number = strings.TrimSpace(number)
	intNumber, _ := strconv.Atoi(number)
	for counter := 1; counter <= intNumber; counter++ {
		cube = intNumber * intNumber * intNumber
	}
	fmt.Println("Cube of a",intNumber, "number:",cube)

}

Explanation:

In this program, we calculating the cube of a number.
As an input we take a number from user.
As an output we calculating the cube of a number that user enter.
