Write a program to store temperature of two cities of a week and display it.

        
            package main

import (
	"bufio"
	"fmt"
	"os"
	"strconv"
	"strings"
)

func main() {
	reader := bufio.NewReader(os.Stdin)
	const city int = 2
	const week int = 7
	var tempr [city][week]int
	for i := 0; i < city; i++ {
		for j := 0; j < week; j++ {
			fmt.Println("City", i+1,",", "Day", j+1,":")
			temp, _ := reader.ReadString('\n')
			temp = strings.TrimSpace(temp)
			intTemp, _ := strconv.Atoi(temp)
			tempr[i][j] = intTemp
		}
	}
	fmt.Println("Displaying Values:")
	for i := 0; i < city; i++ {
		for j := 0; j < week; j++ {
			fmt.Println("City", i+1,",", "Day", j+1,"=", tempr[i][j])

		}
	}
}
