package main

import (
	"bufio"
	"fmt"
	"os"
	"strconv"
	"strings"
)

func main() {
	reader := bufio.NewReader(os.Stdin)
	fmt.Println("Are you ready to perform a dare? YES / NO")
	dare, _ := reader.ReadString('\n')
	dare = strings.TrimSpace(dare)
	dare = strings.ToUpper(dare)
	// fmt.Println("Value of dare is: " + dare + ".")
	if dare == "NO" {
		fmt.Println("Have A Wonderfull Day")
	}
	// if dare == "Yes\r\n" {
	// 	fmt.Println("Choose any number and then see your dare and do it.So,Are you ready? ")
	// }
	
	if dare == "YES" {
		fmt.Println("Choose any number and then see your dare and do it.So,Are you ready? ")
	}
	fmt.Println("GAME IS START")
	fmt.Println("3 2 1 Go .....")
	fmt.Println("Follow the below instruction while choosing the number:")
	fmt.Println("1) Number is no start with zero.")
	fmt.Println("2) Number is no start with any special character symbol like !,@,#,$...")
	fmt.Println("3) Number is no start with alphabets.")
	fmt.Println("Choose number:")
	number, _ := reader.ReadString('\n')
	number = strings.TrimSpace(number)
	intNumber, _ := strconv.Atoi(number)
	for counter := intNumber; counter <= intNumber; counter++ {
		fmt.Println("Number choosed by you:", intNumber)
		if intNumber == 0 {
			break
		}
	}
	fmt.Println("Loading ......")
	fmt.Println(intNumber)
	if intNumber%2 == 0 {
		fmt.Println("Ahm... Your dare is to perform a belly dance/ sketch/ sing a song.")
	} else {
		fmt.Println("Ahm... Your dare is to Write a program in goLanuage which take two number from user and print the multipilcation of that two number.")
	}
	fmt.Println("Are you enjoy this dare? Yes / No:")
	enjoy, _ := reader.ReadString('\n')
	enjoy = strings.TrimSpace(enjoy)
	enjoy =strings.ToUpper(enjoy)
	if enjoy == "YES" {
		fmt.Println("If you like this dare game. Please share your feedback in comment box.")
	}
	if enjoy == "NO" {
		fmt.Println("If you don`t like this dare game then please tell me how can i improve this game. Please share you idea`s with me in comment box.")
	}
	fmt.Println("Thank you. Have a Wonderful day. #stayhome #staywith #codewithme")
}

Explanation:

Some errors are coming, when I make this dare game:

1) If the user is entering the lowercase or mixed case string, it fails to match with "Yes", so I have converted the user input into uppercase and now matching with "YES" to make it correct result.

2) By default, Readstring reads a new line charcater, so to remove it I have used TrimSpaces function, present in strings package
