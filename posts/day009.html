goLanguage Program - Using Switch case statement


"Write a program which input two number and find either sum or difference or multiplication or division of the two numbers depending upon user choice."

            
package main

import (
	"bufio"
	"fmt"
	"os"
	"strconv"
	"strings"
)

func main() {
	reader := bufio.NewReader(os.Stdin)
	fmt.Println("Enter a number1:")
	number1, _ := reader.ReadString('\n')
	number1 = strings.TrimSpace(number1)
	intNumber1, _ := strconv.Atoi(number1)
	fmt.Println("Enter number2:")
	number2, _ := reader.ReadString('\n')
	number2 = strings.TrimSpace(number2)
	intNumber2, _ := strconv.Atoi(number2)
	fmt.Println("Numbers Entered by user:", intNumber1, intNumber2)
	fmt.Println("Press 1 for addition")
	fmt.Println("Press 2 for subtraction")
	fmt.Println("Press 3 for multiply")
	fmt.Println("Press 4 for division")
	press, _ := reader.ReadString('\n')
	press = strings.TrimSpace(press)
	intPressnumber, _ := strconv.Atoi(press)
	fmt.Println("Number press by user:", intPressnumber)
	switch intPressnumber {
	case 1:
		fmt.Println("addition of two number:", intNumber1+intNumber2)
	case 2:
		fmt.Println("subtraction of two number:", intNumber1-intNumber2)
	case 3:
		fmt.Println("multiplication of two number:", intNumber1*intNumber2)
	case 4:
		fmt.Println("division of two number:", intNumber1/intNumber2)
	default:
		fmt.Println("please press valid number from 1-4!")
	}

}

Explanation of the code:

In this we take two input number from user i.e. number1 , number2.
After that based on the user choice we have to perform addition, subtraction, multiplication, division on the two numbers i.e. number1, number2 using switch case.

Switch Case syntax:

				
					Switch expression {
						Case label 1:
						Statement 
						Case label 2:
						Statement 
						……
						……….
						Case label n:
						Statement 
						default: statements
					}

So, in our program:-
Expression: intPressnumber
Case 1: addition
Case 2: subtraction
Case 3: multiplication
Case 4: division
