Program 4 - Array

Write a progam to take 10 integer inputs from user and store them in an array and print them on screen.

        
            package main

            import (
                "bufio"
                "fmt"
                "os"
                "strconv"
                "strings"
            )
            
            func main() {
                reader := bufio.NewReader(os.Stdin)
                var integerNumber [10]int
                for counter := 0; counter < 10; counter++ {
                    fmt.Println( "Integer number",counter+1,":")
                    numbers, _ := reader.ReadString('\n')
                    numbers = strings.TrimSpace(numbers)
                    intNumbers, _ := strconv.Atoi(numbers)
                    integerNumber[counter] = intNumbers
                }
                fmt.Println("10 interger numbers are:",integerNumber)
            }
